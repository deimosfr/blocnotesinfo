    ConfigObject => $ConfigObject,
    LogObject    => $LogObject,

);

Encode::encode( 'UTF-8', $subject ) or die("Cannot encode subject into UTF-8");

    Queue			=> 'queue',
    Lock			=> 'unlock',

    State			=> 'new',

    Subject     => $subject,           # required
    Body        => 'the message text',                 # required
    Charset     => 'UTF-8',
    MimeType    => 'text/plain',
    HistoryType => 'OwnerUpdate',
    HistoryComment => 'Some free text!',
    UserID         => 1,
);
